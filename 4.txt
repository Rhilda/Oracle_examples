1. Получить город, в котором сотрудники суммарно зарабатывают меньше всех.  

select city from (
    select loc.city, sum(emp.salary) s from employees emp
    join departments dep
    on emp.department_id = dep.department_id
    join locations loc
    on loc.location_id = dep.location_id
    GROUP BY (loc.city)
    ORDER BY s
    )
where rownum = 1;

2. Получить список сотрудников, менеджеры которых устроились на работу в 1999 году,
 но при это сами эти сотрудники устроились на работу до 1999.  

select * from employees emp
join employees man
on emp.manager_id = man.employee_id and EXTRACT(YEAR from emp.hire_date) < 1999 and
EXTRACT(YEAR from man.hire_date) = 1999;

3. Дана таблица:
+----+------------+-------------+
| id | recordDate | Temperature |
+----+------------+-------------+
| 1  | 2021-03-01 | 10          |
| 2  | 2021-03-02 | 25          |
| 3  | 2021-03-03 | 20          |
| 4  | 2021-03-04 | 30          |
+----+------------+-------------+
Вывести ID дат с более высокой температурой по сравнению с предыдущими датами.
Например, для приведенной таблицы должны вернуться следующие строки: 
+----+
| id |
+----+
| 2  |
| 4  |
+----+

with ex as ( -- предположим, что department_id это типа температура
    select ROWNUM id, job.department_id Temperature from job_history job
    where ROWNUM < 9
)
select two.id from ex one
join ex two
on one.id = two.id - 1 and
   one.temperature < two.temperature
;

4. Имеется таблица, содержащая два столбца: X и Y. Две пары (X 1 , Y 1) и (X 2 , Y 2) называются 
симметричными парами, если X 1 = Y 2 и X 2 = Y 1.
Напишите запрос для вывода всех таких симметричных пар в порядке возрастания величины X.

Пример: 
Sample Input 
X	Y
20	20
20	20
20	21
23	22
22	23
21	20

Sample Output 
20 20
20 21
22 23

with ex as (
    select 20 X, 20 Y from dual
    union all
    select 20, 20 from dual
    union all
    select 20, 21 from dual
    union all
    select 23, 22 from dual
    union all
    select 22, 23 from dual
    union all
    select 21, 20 from dual
)
select DISTINCT one.* from ex one
join ex two
on one.x = two.y and one.y = two.x
where one.x <= one.y
order by one.x
